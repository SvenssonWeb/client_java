001package io.prometheus.client.logback;
002
003import ch.qos.logback.classic.Level;
004import ch.qos.logback.classic.spi.ILoggingEvent;
005import ch.qos.logback.core.UnsynchronizedAppenderBase;
006import io.prometheus.client.CollectorRegistry;
007import io.prometheus.client.Counter;
008
009public class InstrumentedAppender extends UnsynchronizedAppenderBase<ILoggingEvent> {
010  public static final String COUNTER_NAME = "logback_appender_total";
011  
012  private static final Counter defaultCounter = Counter.build().name(COUNTER_NAME)
013          .help("Logback log statements at various log levels")
014          .labelNames("level")
015          .register();
016  private final Counter.Child traceCounter;
017  private final Counter.Child debugCounter;
018  private final Counter.Child infoCounter;
019  private final Counter.Child warnCounter;
020  private final Counter.Child errorCounter;
021
022  /**
023   * Create a new instrumented appender using the default registry.
024   */
025  public InstrumentedAppender() {
026    this(defaultCounter);
027  }
028
029  /**
030   * Create a new instrumented appender using the supplied registry.
031   */
032  public InstrumentedAppender(CollectorRegistry registry) {
033    this(Counter.build().name(COUNTER_NAME)
034            .help("Logback log statements at various log levels")
035            .labelNames("level")
036            .register(registry));
037  }
038
039  private InstrumentedAppender(Counter counter) {
040    this.traceCounter = counter.labels("trace");
041    this.debugCounter = counter.labels("debug");
042    this.infoCounter = counter.labels("info");
043    this.warnCounter = counter.labels("warn");
044    this.errorCounter = counter.labels("error");
045  }
046
047  @Override
048  public void start() {
049    super.start();
050  }
051
052  @Override
053  protected void append(ILoggingEvent event) {
054    switch (event.getLevel().toInt()) {
055      case Level.TRACE_INT:
056        this.traceCounter.inc();
057        break;
058      case Level.DEBUG_INT:
059        this.debugCounter.inc();
060        break;
061      case Level.INFO_INT:
062        this.infoCounter.inc();
063        break;
064      case Level.WARN_INT:
065        this.warnCounter.inc();
066        break;
067      case Level.ERROR_INT:
068        this.errorCounter.inc();
069        break;
070      default:
071        break;
072    }
073  }
074}
