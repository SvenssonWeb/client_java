001package io.prometheus.client.exporter;
002
003import java.io.IOException;
004import java.net.HttpURLConnection;
005import java.net.URL;
006
007public class DefaultHttpConnectionFactory implements HttpConnectionFactory {
008    @Override
009    public HttpURLConnection create(String url) throws IOException {
010        return (HttpURLConnection) new URL(url).openConnection();
011    }
012}
