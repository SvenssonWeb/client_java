001package io.prometheus.client.jetty;
002
003import io.prometheus.client.Collector;
004import java.util.ArrayList;
005import java.util.Arrays;
006import java.util.Collections;
007import java.util.List;
008import org.eclipse.jetty.server.handler.StatisticsHandler;
009
010/**
011 * Collect metrics from jetty's org.eclipse.jetty.server.handler.StatisticsHandler.
012 * <pre>
013 * Server server = new Server(8080);
014 *
015 * ServletContextHandler context = new ServletContextHandler();
016 * context.setContextPath("/");
017 * server.setHandler(context);
018 *
019 * HandlerCollection handlers = new HandlerCollection();
020 *
021 * StatisticsHandler statisticsHandler = new StatisticsHandler();
022 * statisticsHandler.setServer(server);
023 * handlers.addHandler(statisticsHandler);
024 *
025 * // Register collector.
026 * new JettyStatisticsCollector(statisticsHandler).register();
027 *
028 * server.setHandler(handlers);
029 *
030 * server.start();
031 * </pre>
032 */
033public class JettyStatisticsCollector extends Collector {
034  private final StatisticsHandler statisticsHandler;
035  private static final List<String> EMPTY_LIST = new ArrayList<String>();
036
037  public JettyStatisticsCollector(StatisticsHandler statisticsHandler) {
038    this.statisticsHandler = statisticsHandler;
039  }
040
041  @Override
042  public List<MetricFamilySamples> collect() {
043    return Arrays.asList(
044            buildCounter("jetty_requests_total", "Number of requests", statisticsHandler.getRequests()),
045            buildGauge("jetty_requests_active", "Number of requests currently active", statisticsHandler.getRequestsActive()),
046            buildGauge("jetty_requests_active_max", "Maximum number of requests that have been active at once", statisticsHandler.getRequestsActiveMax()),
047            buildGauge("jetty_request_time_max_seconds", "Maximum time spent handling requests", statisticsHandler.getRequestTimeMax() / 1000.0),
048            buildCounter("jetty_request_time_seconds_total", "Total time spent in all request handling", statisticsHandler.getRequestTimeTotal() / 1000.0),
049            buildCounter("jetty_dispatched_total", "Number of dispatches", statisticsHandler.getDispatched()),
050            buildGauge("jetty_dispatched_active", "Number of dispatches currently active", statisticsHandler.getDispatchedActive()),
051            buildGauge("jetty_dispatched_active_max", "Maximum number of active dispatches being handled", statisticsHandler.getDispatchedActiveMax()),
052            buildGauge("jetty_dispatched_time_max", "Maximum time spent in dispatch handling", statisticsHandler.getDispatchedTimeMax()),
053            buildCounter("jetty_dispatched_time_seconds_total", "Total time spent in dispatch handling", statisticsHandler.getDispatchedTimeTotal() / 1000.0),
054            buildCounter("jetty_async_requests_total", "Total number of async requests", statisticsHandler.getAsyncRequests()),
055            buildGauge("jetty_async_requests_waiting", "Currently waiting async requests", statisticsHandler.getAsyncRequestsWaiting()),
056            buildGauge("jetty_async_requests_waiting_max", "Maximum number of waiting async requests", statisticsHandler.getAsyncRequestsWaitingMax()),
057            buildCounter("jetty_async_dispatches_total", "Number of requested that have been asynchronously dispatched", statisticsHandler.getAsyncDispatches()),
058            buildCounter("jetty_expires_total", "Number of async requests requests that have expired", statisticsHandler.getExpires()),
059            buildStatusCounter(),
060            buildGauge("jetty_stats_seconds", "Time in seconds stats have been collected for", statisticsHandler.getStatsOnMs() / 1000.0),
061            buildCounter("jetty_responses_bytes_total", "Total number of bytes across all responses", statisticsHandler.getResponsesBytesTotal())
062    );
063  }
064
065  private static MetricFamilySamples buildGauge(String name, String help, double value) {
066    return new MetricFamilySamples(
067            name,
068            Type.GAUGE,
069            help,
070            Collections.singletonList(new MetricFamilySamples.Sample(name, EMPTY_LIST, EMPTY_LIST, value)));
071  }
072
073  private static MetricFamilySamples buildCounter(String name, String help, double value) {
074    return new MetricFamilySamples(
075        name,
076        Type.COUNTER,
077        help,
078        Collections.singletonList(new MetricFamilySamples.Sample(name, EMPTY_LIST, EMPTY_LIST, value)));
079  }
080
081  private MetricFamilySamples buildStatusCounter() {
082    String name = "jetty_responses_total";
083    return new MetricFamilySamples(
084            name,
085            Type.COUNTER,
086            "Number of requests with response status",
087            Arrays.asList(
088                    buildStatusSample(name, "1xx", statisticsHandler.getResponses1xx()),
089                    buildStatusSample(name, "2xx", statisticsHandler.getResponses2xx()),
090                    buildStatusSample(name, "3xx", statisticsHandler.getResponses3xx()),
091                    buildStatusSample(name, "4xx", statisticsHandler.getResponses4xx()),
092                    buildStatusSample(name, "5xx", statisticsHandler.getResponses5xx())
093            )
094    );
095  }
096
097  private static MetricFamilySamples.Sample buildStatusSample(String name, String status, double value) {
098    return new MetricFamilySamples.Sample(
099            name,
100            Collections.singletonList("code"),
101            Collections.singletonList(status),
102            value);
103  }
104}
