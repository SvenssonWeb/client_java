001package io.prometheus.client.bridge;
002
003import io.prometheus.client.Collector;
004import io.prometheus.client.CollectorRegistry;
005
006import java.io.BufferedWriter;
007import java.io.IOException;
008import java.io.OutputStreamWriter;
009import java.io.PrintWriter;
010import java.net.Socket;
011import java.nio.charset.Charset;
012import java.util.Collections;
013import java.util.logging.Level;
014import java.util.logging.Logger;
015import java.util.regex.Pattern;
016import java.util.regex.Matcher;
017
018/**
019 * Export metrics in the Graphite plaintext format.
020 * <p>
021 * <pre>
022 * {@code
023 *  Graphite g = new Graphite("localhost", 2003);
024 *  // Push the default registry once.
025 *  g.push(CollectorRegistry.defaultRegistry);
026 *
027 *  // Push the default registry every 60 seconds.
028 *  Thread thread = g.start(CollectorRegistry.defaultRegistry, 60);
029 *  // Stop pushing.
030 *  thread.interrupt();
031 *  thread.join();
032 * }
033 * </pre>
034 * <p>
035 */
036public class Graphite {
037  private static final Logger logger = Logger.getLogger(Graphite.class.getName());
038
039  private final String host;
040  private final int port;
041  private static final Pattern INVALID_GRAPHITE_CHARS = Pattern.compile("[^a-zA-Z0-9_-]");
042  /**
043   * Construct a Graphite Bridge with the given host:port.
044   */
045  public Graphite(String host, int port) {
046    this.host = host;
047    this.port = port;
048  }
049
050  /**
051   * Push samples from the given registry to Graphite.
052   */
053  public void push(CollectorRegistry registry) throws IOException {
054    Socket s = new Socket(host, port);
055    BufferedWriter writer = new BufferedWriter(new PrintWriter(new OutputStreamWriter(s.getOutputStream(), Charset.forName("UTF-8"))));
056    Matcher m = INVALID_GRAPHITE_CHARS.matcher("");
057    long now = System.currentTimeMillis() / 1000;
058    for (Collector.MetricFamilySamples metricFamilySamples: Collections.list(registry.metricFamilySamples())) {
059      for (Collector.MetricFamilySamples.Sample sample: metricFamilySamples.samples) {
060        m.reset(sample.name);
061        writer.write(m.replaceAll("_"));
062        for (int i = 0; i < sample.labelNames.size(); ++i) {
063          m.reset(sample.labelValues.get(i));
064          writer.write("." + sample.labelNames.get(i) + "." + m.replaceAll("_"));
065        }
066        writer.write(" " + sample.value + " " + now + "\n");
067      }
068    }
069    writer.close();
070    s.close();
071  }
072
073  /**
074   * Push samples from the given registry to Graphite every minute.
075   */
076  public Thread start(CollectorRegistry registry) {
077    return start(registry, 60);
078  }
079
080  /**
081   * Push samples from the given registry to Graphite at the given interval.
082   */
083  public Thread start(CollectorRegistry registry, int intervalSeconds) {
084    Thread thread = new PushThread(registry, intervalSeconds);
085    thread.setDaemon(true);
086    thread.start();
087    return thread;
088  }
089
090  private class PushThread extends Thread {
091    private final CollectorRegistry registry;
092    private final int intervalSeconds;
093
094    PushThread(CollectorRegistry registry, int intervalSeconds) {
095      this.registry = registry;
096      this.intervalSeconds = intervalSeconds;
097    }
098
099    public void run() {
100      long waitUntil = System.currentTimeMillis();
101      while (true) {
102        try {
103          push(registry);
104        } catch (IOException e) {
105          logger.log(Level.WARNING, "Exception " + e + " pushing to " + host + ":" + port, e);
106        }
107
108        long now = System.currentTimeMillis();
109        // We may skip some pushes if we're falling behind.
110        while (now >= waitUntil) {
111          waitUntil += intervalSeconds * 1000;
112        }
113        try {
114          Thread.sleep(waitUntil - now);
115        } catch (InterruptedException e) {
116          return;
117        }
118      }
119    }
120  }
121}
